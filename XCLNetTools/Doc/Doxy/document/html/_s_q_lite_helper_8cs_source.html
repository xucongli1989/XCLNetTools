458  public static DataTable SelectPaging(string connString, string tableName, string strColumns, string strWhere, string strOrder, int pageSize, int currentIndex, out int recordOut)
459  {
460  DataTable dt = new DataTable();
461  recordOut = Convert.ToInt32(ExecuteScalar(connString, "select count(*) from " + tableName, CommandType.Text));
462  string pagingTemplate = "select {0} from {1} where {2} order by {3} limit {4} offset {5} ";
463  int offsetCount = (currentIndex - 1) * pageSize;
464  string commandText = String.Format(pagingTemplate, strColumns, tableName, strWhere, strOrder, pageSize.ToString(), offsetCount.ToString());
465  using (DbDataReader reader = ExecuteReader(connString, commandText, CommandType.Text))
466  {
467  if (reader != null)
468  {
469  dt.Load(reader);
470  }
471  }
472  return dt;
473  }

485  private static void PrepareCommand(SQLiteCommand cmd, SQLiteConnection conn, ref SQLiteTransaction trans, bool useTrans, CommandType cmdType, string cmdText, params SQLiteParameter[] cmdParms)
486  {
487  if (conn.State != ConnectionState.Open)
488  conn.Open();
489 
490  cmd.Connection = conn;
491  cmd.CommandText = cmdText;
492 
493  if (useTrans)
494  {
495  trans = conn.BeginTransaction(IsolationLevel.ReadCommitted);
496  cmd.Transaction = trans;
497  }
498 
499  cmd.CommandType = cmdType;
500 
501  if (cmdParms != null)
502  {
503  foreach (SQLiteParameter parm in cmdParms)
504  cmd.Parameters.Add(parm);
505  }
506  }
507  }
