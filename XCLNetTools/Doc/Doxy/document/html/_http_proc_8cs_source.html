284  private byte[] GetData(HttpWebRequest request)
285  {
286  HttpWebResponse response = (HttpWebResponse)request.GetResponse();
287  Stream stream = response.GetResponseStream();
288  responseHeaders = response.Headers;
289  //SaveCookiesToDisk();
290 
291  DownloadEventArgs args = new DownloadEventArgs();
292  if (responseHeaders[HttpResponseHeader.ContentLength] != null)
293  args.TotalBytes = Convert.ToInt32(responseHeaders[HttpResponseHeader.ContentLength]);
294 
295  MemoryStream ms = new MemoryStream();
296  int count = 0;
297  byte[] buf = new byte[bufferSize];
298  while ((count = stream.Read(buf, 0, buf.Length)) > 0)
299  {
300  ms.Write(buf, 0, count);
301  if (this.DownloadProgressChanged != null)
302  {
303  args.BytesReceived += count;
304  args.ReceivedData = new byte[count];
305  Array.Copy(buf, args.ReceivedData, count);
306  this.DownloadProgressChanged(this, args);
307  }
308  }
309  stream.Close();
310  //解压
311  if (ResponseHeaders[HttpResponseHeader.ContentEncoding] != null)
312  {
313  MemoryStream msTemp = new MemoryStream();
314  count = 0;
315  buf = new byte[100];
316  switch (ResponseHeaders[HttpResponseHeader.ContentEncoding].ToLower())
317  {
318  case "gzip":
319  GZipStream gzip = new GZipStream(ms, CompressionMode.Decompress);
320  while ((count = gzip.Read(buf, 0, buf.Length)) > 0)
321  {
322  msTemp.Write(buf, 0, count);
323  }
324  return msTemp.ToArray();
325 
326  case "deflate":
327  DeflateStream deflate = new DeflateStream(ms, CompressionMode.Decompress);
328  while ((count = deflate.Read(buf, 0, buf.Length)) > 0)
329  {
330  msTemp.Write(buf, 0, count);
331  }
332  return msTemp.ToArray();
333 
334  default:
335  break;
336  }
337  }
338  return ms.ToArray();
339  }

346  private void PostData(HttpWebRequest request, byte[] postData)
347  {
348  int offset = 0;
349  int sendBufferSize = bufferSize;
350  int remainBytes = 0;
351  Stream stream = request.GetRequestStream();
352  UploadEventArgs args = new UploadEventArgs();
353  args.TotalBytes = postData.Length;
354  while ((remainBytes = postData.Length - offset) > 0)
355  {
356  if (sendBufferSize > remainBytes) sendBufferSize = remainBytes;
357  stream.Write(postData, offset, sendBufferSize);
358  offset += sendBufferSize;
359  if (this.UploadProgressChanged != null)
360  {
361  args.BytesSent = offset;
362  this.UploadProgressChanged(this, args);
363  }
364  }
365  stream.Close();
366  }

373  private HttpWebRequest CreateRequest(string url, string method)
374  {
375  Uri uri = new Uri(url);
376 
377  if (uri.Scheme == "https")
378  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback(this.CheckValidationResult);
379 
380  // Set a default policy level for the "http:" and "https" schemes.
381  HttpRequestCachePolicy policy = new HttpRequestCachePolicy(HttpRequestCacheLevel.Revalidate);
382  HttpWebRequest.DefaultCachePolicy = policy;
383 
384  HttpWebRequest request = (HttpWebRequest)WebRequest.Create(uri);
385  request.AllowAutoRedirect = false;
386  request.AllowWriteStreamBuffering = false;
387  request.Method = method;
388  if (proxy != null)
389  request.Proxy = proxy;
390  request.CookieContainer = cc;
391  foreach (string key in requestHeaders.Keys)
392  {
393  request.Headers.Add(key, requestHeaders[key]);
394  }
395  requestHeaders.Clear();
396  return request;
