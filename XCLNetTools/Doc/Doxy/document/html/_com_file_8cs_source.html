146  public static void DownLoadFile(string path, string realName)
147  {
148  path = ComFile.MapPath(path);
149  realName = string.IsNullOrEmpty(realName) ? "文件下载" : realName;
150  FileInfo info = new FileInfo(path);
151  HttpResponse Response = HttpContext.Current.Response;
152  Response.Clear();
153  Response.ClearHeaders();
154  Response.Buffer = false;
155  Response.ContentType = "application/octet-stream";
156  string exp = HttpContext.Current.Request.UserAgent.ToLower();
157  if (exp.IndexOf("msie") > -1)
158  {
159  //当客户端使用IE时，对其进行编码；We should encode the filename when our visitors use IE
160  //使用 ToHexString 代替传统的 UrlEncode()；We use "ToHexString" replaced "context.Server.UrlEncode(fileName)"
161  realName = XCLNetTools.Encode.Hex.ToHexString(realName);
162  }
163  Response.AppendHeader("Content-Disposition", "attachment;filename=" + realName.Replace("+", "_").Replace(" ", ""));
164  Response.AppendHeader("Content-Length", info.Length.ToString());
165  Response.WriteFile(path);
166  Response.Flush();
167  Response.End();
168  }
