523  private static string Rep_date(Match mc)
524  {
525  const string cnd = "〇一二三四五六七八九";
526  string val = mc.Value;
527  string digit = val.Substring(0, val.Length - 1);
528  char c = val[val.Length - 1];
529  val = "";
530  switch (c)
531  {
532  case '年':
533  foreach (char s in digit) val += cnd[int.Parse(s.ToString())];
534  break;
535 
536  case '月':
537  case '日':
538  if (digit.Length == 1)
539  val += cnd[int.Parse(digit)];
540  else
541  {
542  val += cnd[int.Parse(digit[0].ToString())] + "十";
543  val = val.TrimStart('一');
544  val += cnd[int.Parse(digit[1].ToString())].ToString().Trim('〇');
545  }
546  break;
547 
548  default:
549  return "格式错误";
550  }
