41  public static List<XCLNetTools.Entity.KeyValue> GetCookieList(string cookie)
42  {
43  List<XCLNetTools.Entity.KeyValue> cookielist = new List<XCLNetTools.Entity.KeyValue>();
44  foreach (string item in cookie.Split(new string[] { ";", "," }, StringSplitOptions.RemoveEmptyEntries))
45  {
46  if (Regex.IsMatch(item, @"([\s\S]*?)=([\s\S]*?)$"))
47  {
48  Match m = Regex.Match(item, @"([\s\S]*?)=([\s\S]*?)$");
49  cookielist.Add(new XCLNetTools.Entity.KeyValue() { Key = m.Groups[1].Value, Value = m.Groups[2].Value });
50  }
51  }
52  return cookielist;
53  }
