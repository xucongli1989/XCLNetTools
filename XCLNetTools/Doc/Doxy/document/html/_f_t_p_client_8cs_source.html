808  private Socket CreateDataSocket()
809  {
810  SendCommand("PASV");
811  if (iReplyCode != 227)
812  {
813  throw new IOException(strReply.Substring(4));
814  }
815  int index1 = strReply.IndexOf('(');
816  int index2 = strReply.IndexOf(')');
817  string ipData = strReply.Substring(index1 + 1, index2 - index1 - 1);
818  int[] parts = new int[6];
819  int len = ipData.Length;
820  int partCount = 0;
821  string buf = "";
822  for (int i = 0; i < len && partCount <= 6; i++)
823  {
824  char ch = Char.Parse(ipData.Substring(i, 1));
825  if (Char.IsDigit(ch))
826  buf += ch;
827  else if (ch != ',')
828  {
829  throw new IOException("Malformed PASV strReply: " + strReply);
830  }
831  if (ch == ',' || i + 1 == len)
832  {
833  try
834  {
835  parts[partCount++] = Int32.Parse(buf);
836  buf = "";
837  }
838  catch (Exception)
839  {
840  throw new IOException("Malformed PASV strReply: " + strReply);
841  }
842  }
843  }
844  string ipAddress = parts[0] + "." + parts[1] + "." + parts[2] + "." + parts[3];
845  int port = (parts[4] << 8) + parts[5];
846  Socket s = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
847  IPEndPoint ep = new IPEndPoint(IPAddress.Parse(ipAddress), port);
848  try
849  {
850  s.Connect(ep);
851  }
852  catch (Exception)
853  {
854  throw new IOException("无法连接ftp服务器");
855  }
856  return s;
857  }
